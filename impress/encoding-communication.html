type Server[T, S] = Channel[Req[T, S]]

def server(f: T => S) = {
  val c = open[Req[T, S]]
  c.events onEvent {
    case (x, ch) => ch ! f(x)
  }
  c.channel
}

type TwoWay[I, O] = (Channel[I], Events[O])

def twoWayServer[I, O]:
  (TwoWay.Server[I, O], Events[TwoWay[O, I]]) = {
  val c = open[TwoWay.Req[I, O]]
  val connections = c.events map {
    case (input, reply) =>
      val output = open[O]
      reply ! output.channel
      (input, output.events)
  }
  (c.channel, connections)
}
        
      
        
def connect[I, O](
  s: TwoWay.Server[I, O]
): Events[TwoWay[I, O]] = {
  val c = open[I]
  (s ? c) map { output =>
    (output, c.events)
  }
}
